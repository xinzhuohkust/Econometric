make_chart <- function(country) {
  
  if(length(country) > 1) {
    data_1.1.1 <- data_1.1  %>% 
      filter(country_code %in% country) %>% 
      mutate(country_code = deparse(substitute(country))) %>% 
      group_by(date, country_code) %>% 
      summarise(mean = mean(mean)) %>% 
      ungroup() %>% 
      mutate(valuation = ifelse(mean >= 0, "Positive", "Negative"))
  } else {
    data_1.1.1 <- data_1.1 %>%  
      filter(country_code %in% country) %>%
      mutate(valuation = ifelse(mean >= 0, "Positive", "Negative"))
  }
  
  min_axis <- ifelse(min(data_1.1.1$mean) > 0, 0, min(data_1.1.1$mean)) - 0.02
  max_axis <- ifelse(max(data_1.1.1$mean) < 0, 0, max(data_1.1.1$mean)) + 0.05
  min_date <- min(data_1.1.1$date)
  
  sign <- data_1.1.1 %>% 
    filter(date < ymd("2021-10-15")) %>% 
    pull(mean)
  
  chart <- data_1.1.1 %>%
    ggplot(aes(date, mean)) +
    geom_point(aes(color = valuation), size = 2) +
    geom_line(color = "grey50") +
    geom_hline(yintercept = 0, color = "grey50", linetype = "dashed") +
    #geom_text(x = min_date, y = 0.02, label = "Positive", hjust = 0, color = "#0870A5") +
    #geom_text(x = min_date, y = -0.02, label = "Negative", hjust = 0, color = "#DB444B") +
    scale_color_manual(values = c("Positive" = "#0870A5", "Negative" = "#DB444B")) +
    scale_y_continuous(limits = c(min_axis, max_axis)) +
    guides(color = FALSE) +
    labs(title = ifelse(length(country) > 1, deparse(substitute(country)), country)) +
    theme(plot.background = element_rect(fill = NA, color = NA),
          panel.background = element_rect(fill = NA, color = NA),
          axis.title = element_blank(),
          axis.text = element_text(family = "heebo", size = 10),
          panel.grid.minor = element_blank(),
          panel.grid.major = element_line(color = "grey80", linetype = "dotted"),
          plot.title = element_text(family = "heebo", size = 14)) + 
    geom_hline(yintercept = median(data_1.1$mean), linetype = "dashed", size = 1, color = "#E9C825")
  #geom_text(x = min_date, y = median(data_1$mean) + 0.01, label = "Overall Median", hjust = 0, color = "#0870A5") 
  #geom_vline(xintercept = ymd("2021-11-02"), linetype = "dashed", size = 0.1, color = "#DB444B")
  
  if(min(abs(sign - median(data_1.1$mean))) > 0.055 & mean(sign) >= median(data_1.1$mean)) {
    chart <- chart + 
      geom_text(x = min_date, y = median(data_1.1$mean) + 0.015, label = "Overall Median", hjust = 0, color = "#0870A5")
  } else {
    chart <- chart
  }
  
  if(mean(sign) < median(data_1.1$mean)) {
    chart <- chart + 
      geom_text(x = min_date, y = median(data_1.1$mean) + 0.03, label = "Overall Median", hjust = 0, color = "#0870A5")
  } 
  
  return(chart)
  
} # chart
